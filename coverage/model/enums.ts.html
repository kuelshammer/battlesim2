export function numericCR(cr: ChallengeRating) {
    switch (cr) {
        case '—': return 0;
        case '1/8': return 1 / 8;
        case '1/4': return 1 / 4;
        case '1/2': return 1 / 2;
        default: return Number(cr);
    }
}
